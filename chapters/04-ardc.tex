\section{Cite and credit}
% TODO draft from: swh-ardc.org::#swh-dc. Refs: DiCosmoSEN2020, cise-2020-doi,
% force11citationprinciples, SCIDWG2020.
The \texttt{biblatex-software} style adds four entry types and a \texttt{swhid=}
field~\autocite{DiCosmoSEN2020,cise-2020-doi}. For example, this note cites a specific
version of a library, \autocite{parmap-1.1.1}, and a code fragment within
it, \autocite{simplemapper}.

\begin{handson}[Hands-on: archive and identify your code]
\begin{enumerate}[nosep,leftmargin=*]
  \item Archive your repository at \url{https://save.softwareheritage.org} and wait for the
        visit to reach status \texttt{full}.
  \item Install the identifier tool: \texttt{pip install 'swh.model[cli]'}.
  \item Compute a directory \textsc{swhid}: \texttt{swh identify -{}-type directory ./src}.
  \item Open your file in the archive and copy the \emph{qualified} \textsc{swhid} from the
        \emph{Permalinks} tab. Paste it after
        \texttt{https://archive.softwareheritage.org/} to confirm it resolves.
\end{enumerate}
\end{handson}
